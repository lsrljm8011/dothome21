const sliderWrap = $(".slider-wrap");
let currentIndex = 0;
let sliderDuration = 600;
let sliderEasing = "easeOutExpo";

const sliderImg1 = $(".slider-img1");
const sliderInner1 = $(".slider-inner1"); 
const slider1 = $(".slider1");

let sliderCount1 = slider1.length; 
let sliderHeight1 = slider1.height();

const sliderImg2 = $(".slider-img2");
const sliderInner2 = $(".slider-inner2"); 
const slider2 = $(".slider2");

let sliderHeight2 = slider2.height();

setInterval(function() {
    //sample 1번
    (currentIndex < sliderCount1-1) ? currentIndex++ : currentIndex = 0
    let sliderPosition = currentIndex * -sliderHeight1;
    sliderInner1.animate({ top: sliderPosition }, sliderDuration, sliderEasing);
    
    //sample 2
    sliderInner2.animate({top: -sliderHeight2}, sliderDuration, function(){ 
        $(".slider2:first-child").appendTo(sliderInner2); 
        sliderInner2.css("top", "");  
    });
}, 3000);
