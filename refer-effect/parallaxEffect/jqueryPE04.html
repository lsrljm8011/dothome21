.paraNav {
    top: 20px;
    transition: top 0.4s ease;
}
.paraNav.hide {
    top: -200px;
}
.paraNav li a {
    width: auto;
    padding: 5px 20px;
}
.paraNav li.active a {
    border-radius: 30px;
}

$(".paraNav ul li").click(function(e){
    e.preventDefault();
    let target = $(this);   //몇번째 선택했는지 알 수 있음
    let index = target.index();
    let section = $(".content-item").eq(index); //content-item와 일치하는 것을 찾아서 index를 가져옴
    let offset = section.offset().top;  //index가져온 content-item의 offsettop값을 구함
    $("html, body").animate({scrollTop:offset}, 800, "easeInOutCirc")
});

let nowScrollTop;   //현재 scrollTop값
let lastScrollTop = 0;  //마지막(1초전) scrollTop 값

$(window).scroll(function(){
    let scrollTop = $(window).scrollTop();  //윈도우의 스크롤 탑 값 가져옴, window = document = this

    //버튼용
    $(".content-item").each(function(index){   
        if(scrollTop >= $(this).offset().top -2){   
            $(".paraNav ul li") .eq(index).addClass("active").siblings().removeClass("active"); 
        }
    });

    //출력용
    $(".paraScroll span").text(scrollTop); 

    nowScrollTop = true;

    setInterval(() => {
        if(nowScrollTop) {  
            nowScrollTop = false;   //true가 false로 바뀌기 때문에 1번만 감지함
            hasScroll();    //실행함수
        } 
    }, 500);   //스크롤을 내렸을 때 0.5초에 한번 감지, 올렸을 때 0.5초에 한번 감지 (스크롤을 할 때만 감지함)
}); 

function hasScroll(){
    let scrollTop = $(this).scrollTop();

    if(scrollTop > lastScrollTop){  //현재 scrollTop값이 마지막(1초전) scrollTop값보다 크면
        $(".paraNav").addClass("hide");
    } else {
        $(".paraNav").removeClass("hide");
    }
    lastScrollTop = scrollTop;

    console.log("scrollTop :" + scrollTop)
    console.log("lastScrollTop :" + lastScrollTop)
}
