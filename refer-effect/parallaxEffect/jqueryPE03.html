.paraNav {
    top: -200px;
    transition: top 0.4s ease;
}
.paraNav.show {
    top: 20px;
}
.paraNav li a {
    width: auto;
    padding: 5px 20px;
}
.paraNav li.active a {
    border-radius: 30px;
}

$(".paraNav ul li").click(function(e){  //메뉴를 클릭하면 이벤트 발생
    e.preventDefault();
    let target = $(this);
    let index = target.index();
    let section = $(".content-item").eq(index); 
    let offset = section.offset().top;  //현재 세션의 위치를 오프셋 탑으로 구함
    $("html, body").animate({scrollTop : offset}, 600)  
});

$(window).scroll(function(){
    let scrollTop = $(window).scrollTop();  //윈도우의 스크롤 탑 값 가져옴, window = document = this

    //숨김 메뉴
    // if(scrollTop > 0){
    //     $(".paraNav").addClass("show");
    // } else {
    //     $(".paraNav").removeClass("show");
    // }

    (scrollTop > 0) ? $(".paraNav").addClass("show") : $(".paraNav").removeClass("show");   //삼항연산자

    //버튼용
    $(".content-item").each(function(index){   
        if(scrollTop >= $(this).offset().top -2){   
            $(".paraNav ul li") .eq(index).addClass("active").siblings().removeClass("active"); 
        }
    });

    //출력용
    $(".paraScroll span").text(scrollTop);   
});
