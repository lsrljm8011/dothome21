const scrollReveal = function(){
    const reveal = document.querySelectorAll(".reveal");    //선택자

    reveal.forEach(el => {
        let scrollTop = (window.pageYOffset || document.documentElement.scrollTop || window.scrollY) + window.innerHeight;  //기존의 스크린에서 화면의 height만큼 더함 -> 더 빨리 나옴
        let revealData = el.dataset.delay;

        if(scrollTop >= el.parentElement.offsetTop){
            if(revealData == undefined){
                el.classList.add("show");
            } else {
                setTimeout(() => {
                    el.classList.add("show");
                }, revealData)
            }
        }
        //출력용
        document.querySelector(".paraScroll span").innerText = parseInt(scrollTop);

        //메뉴
        document.querySelectorAll(".content-item").forEach((item, index) => {
            if(scrollTop >= item.offsetTop){
                document.querySelectorAll(".paraNav ul li").forEach(li => {
                    li.classList.remove("active");
                });
                document.querySelector(".paraNav ul li:nth-child(" + (index+1) + ")").classList.add("active");
            }
        });
    });
}

window.addEventListener("scroll", scrollReveal);
