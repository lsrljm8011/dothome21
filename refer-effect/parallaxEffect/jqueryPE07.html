    opacity: 0;
    display: inline-block;
    min-width: 1vw;
    transform: translateY(50px) rotate(30deg) scale(1.5);

//한글자씩 쪼개기(여러개)
$(".content-item .desc").each(function(index){
    let text = $(this).text(); 
    let split = text.split('').join("</span><span area-hidden='true'>"); //area-hidden='true' : 웹표준은 준수하기 위해 생긴 속성, 글자를 읽지 않도록 해줌
        
    split = "<span>"+split+"</span>";   
    $(this).html(split).attr("area-label", text);   //접근성을 위해 p태그 뒤에 속성을 넣어 글자를 읽을 수 읽도록 해줌
});
