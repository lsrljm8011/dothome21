const cursor = document.querySelector(".cursor");
const circle = cursor.getBoundingClientRect();  

function mouseCircle(e) {

    gsap.to(cursor, {duration: 0.5, left: e.pageX -circle.width/2, top: e.pageY -circle.height/2});
    let pageX = e.pageX;
    let pageY = e.pageY;

    //출력
    document.querySelector(".pageX").textContent = pageX;
    document.querySelector(".pageY").textContent = pageY;
}

window.addEventListener("mousemove", mouseCircle);
